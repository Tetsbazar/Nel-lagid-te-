  for(var i=1;i<=9;i++){

function goHub(){
  document.getElementById('hub').style.display = 'block';
  document.getElementById('test').classList.add('hide');
  document.getElementById('res').classList.remove('show');
  document.getElementById('timerBox').classList.remove('danger');
  clearInterval(timer);
}

function startQuiz(id){
  curSet = id;
  curQ = 0; score = 0; wrong = 0; answered = false; timeLeft = 600; userAns = [];
  document.getElementById('hub').style.display = 'none';
  document.getElementById('test').classList.remove('hide');
  document.getElementById('res').classList.remove('show');
  document.getElementById('timerBox').classList.remove('danger');
  var s = sets[id];
  document.getElementById('setName').textContent = s.n;
  document.getElementById('setTopic').textContent = s.t;
  clearInterval(timer);
  timer = setInterval(tick,1000);
  loadQ();
}

function tick(){
  timeLeft--;
  var m=Math.floor(timeLeft/60), s=timeLeft%60;
  document.getElementById('timerDisplay').textContent = m+':'+(s<10?'0':'')+s;
  if(timeLeft<=60) document.getElementById('timerBox').classList.add('danger');
  if(timeLeft<=0){ clearInterval(timer); showResult(); }
}

function loadQ(){
  answered = false;
  var s = sets[curSet];
  var q = s.q[curQ];
  document.getElementById('qnum').textContent = 'প্রশ্ন '+(curQ+1)+' / '+s.q.length;
  document.getElementById('qtext').textContent = q.q;
  document.getElementById('fill').style.width = ((curQ+1)/s.q.length*100)+'%';
  document.getElementById('next').disabled = true;
  document.getElementById('next').textContent = (curQ === s.q.length-1) ? 'ফলাফল দেখো' : 'পরবর্তী →';
  document.getElementById('exp').classList.remove('show');
  var opts = document.getElementById('opts');
  opts.innerHTML = '';
  var labels = ['ক','খ','গ','ঘ'];
  for(var i=0;i<q.o.length;i++){
    var b = document.createElement('div');
    b.className = 'opt';
    b.innerHTML = '<div class="c">'+labels[i]+'</div><span>'+q.o[i]+'</span>';
    b.onclick = (function(idx){return function(){answer(idx);};})(i);
    opts.appendChild(b);
  }
}

function answer(i){
  if(answered) return;
  answered = true;
  var s = sets[curSet];
  var q = s.q[curQ];
  userAns[curQ] = i;
  var btns = document.querySelectorAll('.opt');
  btns.forEach(function(b){b.classList.add('dis');});
  btns[q.a].classList.add('correct');
  if(i !== q.a){ btns[i].classList.add('wrong'); wrong++; } else score++;
  document.getElementById('sc').textContent = 'সঠিক: '+score+' | ভুল: '+wrong;
  document.getElementById('expt').innerHTML = '<strong>ব্যাখ্যা:</strong> '+q.e;
  document.getElementById('exp').classList.add('show');
  document.getElementById('next').disabled = false;
}

function nextQ(){
  var s = sets[curSet];
  curQ++;
  if(curQ >= s.q.length){ clearInterval(timer); showResult(); return; }
  loadQ();
}

function showResult(){
  document.getElementById('test').classList.add('hide');
  var r = document.getElementById('res');
  r.classList.add('show');
  var s = sets[curSet];
  var pct = Math.round(score/s.q.length*100);
  document.getElementById('rsc').textContent = score;
  document.getElementById('rc').textContent = score;
  document.getElementById('rw').textContent = wrong;
  document.getElementById('rp').textContent = pct+'%';
  var grades = [[90,'🌟','অসাধারণ!'], [70,'👍','খুব ভালো!'], [50,'📚','ভালো চেষ্টা!'], [0,'💪','চেষ্টা চালিয়ে যাও!']];
  var g = grades[0];
  for(var i=0;i<grades.length;i++){ if(pct>=grades[i][0]){ g=grades[i]; break; } }
  document.getElementById('ricon').textContent = g[1];
  document.getElementById('rgd').textContent = g[2];
}

function review(){
  document.getElementById('res').classList.remove('show');
  document.getElementById('test').classList.remove('hide');
  curQ = 0;
  loadReview();
}

function loadReview(){
  var s = sets[curSet];
  if(curQ >= s.q.length){ document.getElementById('test').classList.add('hide'); document.getElementById('res').classList.add('show'); return; }
  var q = s.q[curQ];
  document.getElementById('qnum').textContent = 'Review — প্রশ্ন '+(curQ+1)+' / '+s.q.length;
  document.getElementById('qtext').textContent = q.q;
  document.getElementById('fill').style.width = ((curQ+1)/s.q.length*100)+'%';
  document.getElementById('next').disabled = false;
  document.getElementById('next').textContent = (curQ === s.q.length-1) ? 'শেষ করো' : 'পরবর্তী →';
  document.getElementById('next').onclick = function(){ curQ++; loadReview(); };
  var opts = document.getElementById('opts');
  opts.innerHTML = '';
  var labels = ['ক','খ','গ','ঘ'];
  for(var i=0;i<q.o.length;i++){
    var b = document.createElement('div');
    b.className = 'opt dis';
    if(i === q.a) b.classList.add('correct');
    else if(userAns[curQ] === i) b.classList.add('wrong');
    b.innerHTML = '<div class="c">'+labels[i]+'</div><span>'+q.o[i]+'</span>';
    opts.appendChild(b);
  }
  document.getElementById('expt').innerHTML = '<strong>ব্যাখ্যা:</strong> '+q.e;
  document.getElementById('exp').classList.add('show');
  document.getElementById('sc').textContent = 'সবুজ = সঠিক উত্তর';
}
